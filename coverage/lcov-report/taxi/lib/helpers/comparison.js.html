/**
 * Comparison object
 *
 * @constructor
 * @class Comparison
 * @module WebDriver
 * @submodule Helpers
 * @param {Driver} driver
 * @param {string[]} comparisonList List of tools to use for comparison
 */
function Comparison (driver, comparisonList) {
	this._driver = driver;
 
	this._comparisons = {};
	comparisonList.forEach(function (comparison) {
		this._loadComparison(comparison);
	}.bind(this));
}
